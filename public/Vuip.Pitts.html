Introduction

The main theorem, Pitts' uniform interpolation for intuitionistic propositional logic, consists of two parts: 1) the inductive construction of the uniform interpolants; 2) a proof of its correctness.

Throughout the construction and proof, we fix a variable p, with respect to which the uniform interpolant will be computed.

Section UniformInterpolantDefinition.

We define the formulas E\phi and A\phi associated to any formula \phi. This is an implementation of Pitts' Table 5, together with a (mostly automatic) proof that the definition terminates.

(* solves the obligations of the following programs *)
Obligation Tactic := intros; order_tac.

First, the implementation of the rules for calculating E. The names of the rules refer to the table in Pitts' paper. note the use of "lazy" conjunctions, disjunctions and implications
Program Definition e_rule {Δ : env} {ϕ : formula}
  (E : ∀ Δ' (HΔ' : Δ' ≺ Δ), formula)
  (A : ∀ pe (Hpe : pe ≺· (Δ, ϕ)), formula)
  (θ: formula) (Hin : θ ∈ Δ)
   : formula :=
  let Δ' := Δ ∖ {[θ]} in
match θ with
(* E0 *)
| Bot => ⊥
| Var q =>
    if decide (p = q)
    then ⊤ (* default *)
    else E Δ' _ ⊼ q (* E1 *)
(* E2 *)
| δ₁ ∧ δ₂ => E ((Δ'•δ₁)•δ₂) _
(* E3 *)
| δ₁ ∨ δ₂ => E (Δ'•δ₁) _ ⊻ E (Δ' •δ₂) _
| Var q → δ =>
    if decide (p = q)
    then
      if decide (Var p ∈ Δ)
      then E (Δ'•δ) _ (* E5 *)
      else ⊤ (* default *)
    else q ⇢ E (Δ'•δ) _ (* E4 *)
(* E6 *)
| (δ₁ ∧ δ₂)→ δ₃ => E (Δ'•(δ₁ → (δ₂ → δ₃))) _
(* E7 *)
| (δ₁ ∨ δ₂)→ δ₃ => E (Δ'•(δ₁ → δ₃)•(δ₂ → δ₃)) _
(* the E8 rule is the only one calling A *)
| ((δ₁→ δ₂)→ δ₃) =>
    (E (Δ'•(δ₂ → δ₃)) _⇢ A (Δ'•(δ₂ → δ₃), δ₁ → δ₂) _)⇢ E (Δ'•δ₃) _
| Bot → _ => ⊤ (* default value. *)
end.

The implementation of the rules for defining A is separated into two pieces. Referring to Table 5 in Pitts, the definition a_rule handles A1-8 and A10, and the definition a_rule_pointer handles A9 and A11-13.
Program Definition a_rule
  {Δ : env} {ϕ : formula}
  (E : ∀ Δ' (HΔ' : Δ' ≺ Δ), formula)
  (A : ∀ pe (Hpe : pe ≺· (Δ, ϕ)), formula)
  (θ: formula) (Hin : θ ∈ Δ)
   : formula :=
  let Δ' := Δ ∖ {[θ]} in
match θ with
| Var q =>
    if decide (p = q)
    then
      if decide (Var p = ϕ)
      then ⊤ (* A10 *)
      else ⊥ (* default *)
    else A (Δ', ϕ) _ (* A1 *)
(* A2 *)
| δ₁ ∧ δ₂ => A ((Δ'•δ₁)•δ₂, ϕ) _
(* A3 *)
| δ₁ ∨ δ₂ =>
      (E (Δ'•δ₁) _ ⇢ A (Δ'•δ₁, ϕ) _)
  ⊼ (E (Δ'•δ₂) _ ⇢ A (Δ'•δ₂, ϕ) _)
| Var q → δ =>
    if decide (p = q)
    then
      if decide (Var p ∈ Δ)
      then A (Δ'•δ, ϕ) _ (* A5 *)
      else ⊥ (* default *)
    else q ⊼ A (Δ'•δ, ϕ) _ (* A4 *)
(* A6 *)
| (δ₁ ∧ δ₂)→ δ₃ =>
  A (Δ'•(δ₁ → (δ₂ → δ₃)), ϕ) _
(* A7 *)
| (δ₁ ∨ δ₂)→ δ₃ =>
  A ((Δ'•(δ₁ → δ₃))•(δ₂ → δ₃), ϕ) _
(* A8 *)
| ((δ₁→ δ₂)→ δ₃) =>
      (E (Δ'•(δ₂ → δ₃)) _ ⇢ A (Δ'•(δ₂ → δ₃), (δ₁ → δ₂)) _)
  ⊼ A (Δ'•δ₃, ϕ) _
| Bot => ⊥ (* default *)
| Bot → _ => ⊥
end.

Program Definition a_pointer_rule
  {Δ : env} {ϕ : formula}
  (E : ∀ pe (Hpe : pe ≺· (Δ, ϕ)), formula)
  (A : ∀ pe (Hpe : pe ≺· (Δ, ϕ)), formula)
   : formula := match ϕ with
| Var q =>
    if decide (p = q)
    then ⊥ (* default *)
    else Var q (* A9 *)
| ϕ₁ ∧ ϕ₂ => A (Δ, ϕ₁) _ ⊼ A (Δ, ϕ₂) _ (* A11 *)
| ϕ₁ ∨ ϕ₂ => A (Δ, ϕ₁) _ ⊻ A (Δ, ϕ₂) _ (* A12 *)
| ϕ₁→ ϕ₂ => E (Δ•ϕ₁, ϕ₂) _ ⇢ A (Δ•ϕ₁, ϕ₂) _ (* A13 *)
| Bot => ⊥ (* default *)
end.

(* Ideally, we would want to use the "Function" scheme, but it
  "does not support partial application of the function being defined." and we need partial application
  of E and A in in_map, so we resort to using Fix directly *)
Definition EA : (env * formula) -> (formula * formula) :=
  @Fix _ _ wf_pointed_order _
    (λ pe EA,
      let Δ := fst pe in
      let ϕ := snd pe in
      let E := λ d H, fst (EA (d,ϕ) (env_order_singleton_compat ϕ H)) in
      let A := λ pe0 H, snd (EA pe0 H) in
      let E' := λ pe0 H, fst (EA pe0 H) in
      (⋀ (in_map Δ (e_rule E A)),
      ⋁ (in_map Δ (a_rule E A)) ⊻ a_pointer_rule E' A)
     ).

Definition E Δ ϕ := fst (EA (Δ, ϕ)).
Definition A Δ ϕ := snd (EA (Δ, ϕ)).
Definition E' pe := E (fst pe) (snd pe).

Congruence lemmas: if we apply any of e_rule, a_rule, or a_pointer_rule to two equal environments, then they give the same results.
Lemma e_rule_cong Δ ϕ θ (Hin1 : θ ∈ Δ) Hin2 E1 E2 A1 A2:
  (forall Δ' HΔ' HΔ'', E1 Δ' HΔ' = E2 Δ' HΔ'') ->
  (forall pe Hpe Hpe', A1 pe Hpe = A2 pe Hpe' ) ->
  @e_rule Δ ϕ E1 A1 θ Hin1 = @e_rule Δ ϕ E2 A2 θ Hin2.
Proof.
  intros Heq1 Heq2.
  destruct θ; simpl; try (destruct θ1); repeat (destruct decide);
  f_equal; repeat rewrite Heq1; repeat rewrite Heq2; trivial.
  - erewrite Heq1. reflexivity.
  - f_equal; [apply Heq1| apply Heq2].
Qed.

Lemma a_rule_cong Δ ϕ θ Hin1 Hin2 E1 E2 A1 A2 :
  (forall Δ' HΔ' HΔ'', E1 Δ' HΔ' = E2 Δ' HΔ'') ->
  (forall pe Hpe Hpe', A1 pe Hpe = A2 pe Hpe') ->
  @a_rule Δ ϕ E1 A1 θ Hin1 = @a_rule Δ ϕ E2 A2 θ Hin2.
Proof.
  intros Heq1 Heq2. destruct θ; simpl; trivial.
  - case decide; intro; subst; auto.
  - repeat erewrite Heq1; repeat erewrite Heq2. auto.
  - destruct θ1; try (destruct decide); trivial.
    + destruct decide; trivial.
    + erewrite Heq2; eauto.
    + do 2 erewrite Heq2. erewrite Heq1. eauto.
Qed.

Lemma a_pointer_rule_cong Δ ϕ E1 E2 A1 A2 :
  (forall Δ' HΔ' HΔ'', E1 Δ' HΔ' = E2 Δ' HΔ'') ->
  (forall pe Hpe Hpe', A1 pe Hpe = A2 pe Hpe') ->
  @a_pointer_rule Δ ϕ E1 A1 = @a_pointer_rule Δ ϕ E2 A2.
Proof.
  intros Heq1 Heq2.
  destruct ϕ; simpl; repeat (destruct decide); trivial;
  repeat (erewrite Heq1; eauto); repeat (erewrite Heq2; eauto); trivial.
Qed.

(* equation lemma *)
Lemma EA_eq Δ ϕ:
  (E Δ ϕ = ⋀ (in_map Δ (@e_rule Δ ϕ (λ d _, E d ϕ) (λ d _, A (fst d) (snd d))))) /\
  (A Δ ϕ = (⋁ (in_map Δ (@a_rule Δ ϕ (λ d _, E d ϕ) (λ d _, A (fst d) (snd d)))) ⊻
       @a_pointer_rule Δ ϕ (λ d _, E' d) (λ d _, A (fst d) (snd d)))).
Proof.
simpl. unfold E, A, E', EA.
rewrite -> Fix_eq.
- simpl. split; trivial.
- intros Δ' f g Heq. f_equal.
  + f_equal. apply in_map_ext. intros. apply e_rule_cong; intros; f_equal.
    -- simpl. rewrite (proof_irrelevance _ HΔ' HΔ''); now rewrite Heq.
    -- rewrite (proof_irrelevance _ Hpe Hpe'); now rewrite Heq.
  + f_equal.
    * f_equal. apply in_map_ext. intros. apply a_rule_cong; intros.
      -- erewrite (proof_irrelevance _ HΔ' HΔ''). now rewrite Heq.
      -- rewrite (proof_irrelevance _ Hpe Hpe'); now rewrite Heq.
    * apply a_pointer_rule_cong; intros; f_equal.
      -- rewrite (proof_irrelevance _ HΔ' HΔ''); now rewrite Heq.
      -- rewrite (proof_irrelevance _ Hpe Hpe'); now rewrite Heq.
Qed.

Definition E_eq Δ ϕ := proj1 (EA_eq Δ ϕ).
Definition A_eq Δ ϕ := proj2 (EA_eq Δ ϕ).

End UniformInterpolantDefinition.

Correctness

Section Correctness.

This section contains the proof of Proposition 5, the main correctness result, stating that the E- and A-formulas defined above are indeed left and right uniform interpolants of the original formula, respectively.

(i) Variables

Section VariablesCorrect.

In this subsection we prove (i), which states that the variable p no longer occurs in the E- and A-formulas, and that the E- and A-formulas contain no more variables than the original formula.

(a)


Lemma e_rule_vars Δ θ Hin ϕ
  (E0 : ∀ Δ' (HΔ' : Δ' ≺ Δ), formula)
  (A0 : ∀ pe (Hpe : pe ≺· (Δ, ϕ)), formula) x
  (HE : ∀ Δ0 HΔ0, occurs_in x (E0 Δ0 HΔ0) -> x ≠ p ∧ ∃ θ, θ ∈ Δ0 /\ occurs_in x θ)
  (HA : ∀ pe Hpe, occurs_in x (A0 pe Hpe) -> x ≠ p ∧ (occurs_in x pe.2 \/ ∃ θ, θ ∈ pe.1 /\ occurs_in x θ)) :
occurs_in x (e_rule E0 A0 θ Hin) -> x ≠ p ∧ ∃ θ, θ ∈ Δ /\ occurs_in x θ.
Proof.
destruct θ; unfold e_rule.
- case decide.
  + simpl. intros Heq HF; subst. tauto.
  + simpl. intros Hneq Hocc. apply occurs_in_make_conj in Hocc.
      destruct Hocc as [Hocc|Heq]; vars_tac. subst; tauto.
- simpl. tauto.
- vars_tac.
- intro Hocc. apply occurs_in_make_disj in Hocc as [Hocc|Hocc]; vars_tac.
- destruct θ1; try solve[vars_tac].
  + case decide.
    * intro; subst. case decide.
      -- vars_tac.
      -- simpl; tauto.
    * intros Hneq Hocc. apply occurs_in_make_impl in Hocc as [Heq | Hocc]; vars_tac.
      subst. tauto.
  + simpl; tauto.
  + intros Hocc. apply occurs_in_make_impl in Hocc.
      destruct Hocc as [Hocc|Hocc]; [apply occurs_in_make_impl in Hocc as [Hocc|Hocc]|]; vars_tac; subst; tauto.
Qed.

(b)


Lemma a_rule_vars Δ θ Hin ϕ
  (E0 : ∀ Δ' (HΔ' : Δ' ≺ Δ), formula)
  (A0 : ∀ pe (Hpe : pe ≺· (Δ, ϕ)), formula) x
  (HE : ∀ Δ0 HΔ0, occurs_in x (E0 Δ0 HΔ0) -> x ≠ p ∧ ∃ θ, θ ∈ Δ0 /\ occurs_in x θ)
  (HA : ∀ pe Hpe, occurs_in x (A0 pe Hpe) -> x ≠ p ∧ (occurs_in x pe.2 \/ ∃ θ, θ ∈ pe.1 /\ occurs_in x θ)) :
occurs_in x (a_rule E0 A0 θ Hin) -> x ≠ p ∧ (occurs_in x ϕ \/ ∃ θ, θ ∈ Δ /\ occurs_in x θ).
Proof.
destruct θ; unfold a_rule.
- case decide.
  + intro; subst. case decide; simpl; tauto.
  + intros Hneq Hocc. vars_tac.
- simpl. tauto.
- intro Hocc. vars_tac.
- intros Hocc. apply occurs_in_make_conj in Hocc.
  destruct Hocc as [Hocc|Hocc];
  apply occurs_in_make_impl in Hocc; vars_tac; vars_tac.
- destruct θ1; try solve[vars_tac].
  + case decide.
    * intro; subst. case decide.
      -- intros Hp Hocc. vars_tac.
      -- simpl; tauto.
    * intros Hneq Hocc. apply occurs_in_make_conj in Hocc.
       destruct Hocc as [Heq | Hocc]; vars_tac. subst; tauto.
  + intros Hocc. apply occurs_in_make_conj in Hocc.
      destruct Hocc as [Hocc|Hocc].
      * apply occurs_in_make_impl in Hocc; vars_tac; vars_tac.
      * vars_tac.
Qed.

Lemma a_pointer_rule_vars Δ ϕ
  (E0 : ∀ pe (Hpe : pe ≺· (Δ, ϕ)), formula)
  (A0 : ∀ pe (Hpe : pe ≺· (Δ, ϕ)), formula) x
  (HE : ∀ pe Hpe, occurs_in x (E0 pe Hpe) -> x ≠ p ∧ ∃ θ, θ ∈ pe.1 /\ occurs_in x θ)
  (HA : ∀ pe Hpe, occurs_in x (A0 pe Hpe) -> x ≠ p ∧ (occurs_in x pe.2 \/ ∃ θ, θ ∈ pe.1 /\ occurs_in x θ)) :
  occurs_in x (a_pointer_rule E0 A0) -> x ≠ p ∧ (occurs_in x ϕ \/ ∃ θ, θ ∈ Δ /\ occurs_in x θ).
Proof.
destruct ϕ; simpl.
- case decide; simpl; intros; subst; intuition; auto.
- tauto.
- intros Hocc; apply occurs_in_make_conj in Hocc as [Hocc|Hocc]; vars_tac.
- intros Hocc; apply occurs_in_make_disj in Hocc as [Hocc|Hocc]; vars_tac.
- intro Hocc. vars_tac; destruct Hocc; vars_tac; vars_tac.
Qed.

Proposition EA_vars Δ ϕ x:
  (occurs_in x (E Δ ϕ) -> x <> p /\ ∃ θ, θ ∈ Δ /\ occurs_in x θ) /\
  (occurs_in x (A Δ ϕ) -> x <> p /\ (occurs_in x ϕ \/ (∃ θ, θ ∈ Δ /\ occurs_in x θ))).
Proof.
remember (Δ, ϕ) as pe.
replace Δ with pe.1 by now subst.
replace ϕ with pe.2 by now subst. clear Heqpe Δ ϕ. revert pe.
refine (@well_founded_induction _ _ wf_pointed_order _ _).
intros [Δ ϕ] Hind. simpl.
rewrite E_eq, A_eq. simpl.
split.
(* E *)
- intros Hocc. apply variables_conjunction in Hocc as (φ&Hin&Hφ).
  apply in_in_map in Hin as (ψ&Hin&Heq). subst φ.
  apply e_rule_vars in Hφ.
  + trivial.
  + intros Δ0 Hlt Hocc. apply (Hind (Δ0, ϕ)) in Hocc.
    * simpl in *. auto.
    * unfold pointed_env_order; simpl. now apply env_order_singleton_compat.
  + intros; now apply Hind.
(* A *)
- intro Hocc. apply occurs_in_make_disj in Hocc as [Hocc|Hocc].
  (* disjunction *)
  + apply variables_disjunction in Hocc as (φ&Hin&Hφ).
      apply in_in_map in Hin as (ψ&Hin&Heq). subst φ.
      apply a_rule_vars in Hφ; trivial.
    * intros Δ0 Hlt Hocc. apply (Hind (Δ0, ϕ)) in Hocc.
      -- simpl in *. auto.
      -- unfold pointed_env_order; simpl. now apply env_order_singleton_compat.
    * intros pe Hle Hocc. apply Hind in Hocc; trivial.
  (* pointer rule *)
  + apply a_pointer_rule_vars in Hocc.
    * destruct Hocc as [Hneq [Hocc | Hocc]]; vars_tac.
    * vars_tac; apply Hind in H; trivial; tauto.
    * intros pe Hle Hocc'. apply Hind in Hocc'; trivial.
Qed.

End VariablesCorrect.

(ii) Entailment

Section EntailmentCorrect.

In this section we prove (ii), which states that the E- and A-formula are entailed by the original formula and entail the original formula, respectively.

Hint Extern 5 (?a ≺ ?b) => order_tac : proof.
Opaque make_disj.
Opaque make_conj.

Lemma a_rule_spec Δ θ ϕ Hin E0 (A0 : ∀ pe, (pe ≺· (Δ, ϕ)) → formula)
  (HE : forall Δ ϕ, Δ ⊢ E0 Δ ϕ)
  (HA : forall Δ ϕ Hpe, Δ•A0 (Δ, ϕ) Hpe ⊢ ϕ) :
  (Δ•a_rule E0 A0 θ Hin ⊢ ϕ).
Proof with (auto with proof).
destruct θ; simpl; exhibit Hin 1.
- case decide; intro Hp.
  + subst. case_decide; subst; auto with proof.
  + exch 0...
- constructor 2.
- exch 0. apply AndL. exch 1; exch 0...
- apply make_conj_sound_L.
  exch 0. apply OrL; exch 0.
  + apply AndL. apply make_impl_sound_L. exch 0. apply make_impl_sound_L... (* uses impL_weakening *)
  + apply AndL. apply make_impl_sound_L. exch 0. apply make_impl_sound_L. auto with proof.
- destruct θ1.
  + case decide; intro Hp.
    * subst. case decide.
      -- intros Hp.
         assert(Hin' : (p → θ2) ∈ Δ ∖ {[Var p]})
         by (rewrite (gmultiset_disj_union_difference' _ _ Hp) in Hin; ms).
         assert (Hin'' : Var p ∈ Δ ∖ {[p → θ2]}) by now apply in_difference.
         exhibit Hin'' 2. exch 0; exch 1. apply ImpLVar; auto with proof.
         exch 1. exch 0...
         peapply HA. rewrite <- difference_singleton by trivial. reflexivity.
      -- intro; constructor 2.
    * apply make_conj_sound_L. constructor 4. exch 0. exch 1. exch 0. apply ImpLVar.
      exch 0. apply weakening. exch 0...
  + constructor 2.
  + exch 0. apply ImpLAnd. apply make_impl_complete_L2. exch 0...
  + exch 0. apply ImpLOr. exch 1. exch 0...
  + apply make_conj_sound_L. exch 0. apply ImpLImp; exch 0.
      * apply AndL...
      * apply AndL. exch 0. apply weakening, HA.
Qed.

Proposition entail_correct Δ ϕ : (Δ ⊢ E Δ ϕ) * (Δ•A Δ ϕ ⊢ ϕ).
Proof.
remember (Δ, ϕ) as pe.
replace Δ with pe.1 by now subst.
replace ϕ with pe.2 by now subst. clear Heqpe Δ ϕ. revert pe.
refine (@well_founded_induction _ _ wf_pointed_order _ _).
unfold pointed_env_order.
intros (Δ, ϕ) Hind. simpl.
rewrite E_eq, A_eq.
(* uncurry the induction hypothesis for convenience *)
assert (HE := fun d f x=> fst (Hind (d, f) x)).
assert (HA := fun d f x=> snd (Hind (d, f) x)).
simpl in *. clear Hind.
split. {
(* E *)
apply conjunction_R1.
intros φ Hin. apply in_in_map in Hin.
destruct Hin as (ψ&Hin&Heq).
subst.
destruct ψ; unfold e_rule; exhibit Hin 0; auto using HE with proof.
- case decide; intro; subst; simpl; auto using HE with proof.
  apply ImpR, ExFalso.
- destruct ψ1; auto using HE with proof.
  + case decide; intro Hp.
    * subst. case decide; intro Hp.
      -- assert(Hin'' : Var p ∈ Δ ∖ {[p → ψ2]}) by (apply in_difference; trivial; discriminate).
         exhibit Hin'' 1. apply ImpLVar.
         peapply (HE (Δ ∖ {[p → ψ2]}•ψ2) ϕ). auto with proof.
         rewrite <- difference_singleton; trivial.
      -- apply ImpR, ExFalso.
    * apply make_impl_sound_R, ImpR. exch 0. apply ImpLVar. exch 0. apply weakening, HE. order_tac.
  + apply ImpR, ExFalso.
  + remember (Δ ∖ {[(ψ1_1 → ψ1_2) → ψ2]}) as Δ'.
      apply make_impl_sound_R, ImpR. apply make_impl_sound_L. exch 0. apply ImpLImp.
      * exch 0. auto with proof.
      * exch 0. auto with proof.
}
(* A *)
apply make_disj_sound_L, OrL.
- apply disjunction_L. intros φ Hin.
  apply in_in_map in Hin as (φ' & Heq & Hφ'). subst φ.
  apply a_rule_spec; intros; apply HE || apply HA; order_tac.
- destruct ϕ; simpl; auto using HE with proof.
  + case decide; intro; subst; [constructor 2|constructor 1].
  + apply make_conj_sound_L, AndR; apply AndL; auto using HE with proof.
  + apply ImpR. exch 0. apply make_impl_sound_L, ImpL; auto using HE, HA with proof.
      apply weakening. unfold E'. simpl. apply HE. order_tac.
Qed.

End EntailmentCorrect.

Uniformity

Section UniformCorrect.

The proof in this section, which is the most complex part of the argument, shows that the E- and A-formulas constructed above are indeed uniform interpolants, that is, *any* formula entailed by the original formula and using only variables from that formula except p is already a consequence of the E-interpolant, and similarly on the other side for the A-interpolant.
E's second argument is mostly irrelevant and is only there for uniform treatment with A
Lemma E_irr ϕ' Δ ϕ : ϕ <> Var p \/ ϕ' <> Var p-> E Δ ϕ = E Δ ϕ'.
Proof.
remember ((Δ, ϕ) : pointed_env) as pe.
replace Δ with pe.1 by now subst.
replace ϕ with pe.2 by now subst. clear Heqpe Δ ϕ.
  induction pe as [[Γ φ] Hind0] using (well_founded_induction_type wf_pointed_order).
simpl.
assert(Hind : ∀ Γ0 φ0, ((Γ0, φ0) ≺· (Γ, φ)) → φ0 ≠ p \/ ϕ' ≠ p → E Γ0 φ0 = E Γ0 ϕ')
  by exact (fun Γ0 φ0 => (Hind0 (Γ0, φ0))).
intro Hneq. do 2 rewrite E_eq. f_equal. apply in_map_ext.
intros φ' Hin. unfold e_rule.
destruct φ'; repeat rewrite (Hind _ φ) by (trivial; order_tac); trivial.
destruct φ'1; repeat rewrite (Hind _ φ) by (trivial; order_tac); trivial.
Qed.

Lemma E_left {Γ} {θ} {Δ} {φ φ'} (Hin : φ ∈ Δ) :
(Γ•e_rule (λ (d : env) (_ : d ≺ Δ), E d φ') (λ pe (_ : pe ≺· (Δ, φ')), A pe.1 pe.2) φ Hin) ⊢ θ ->
Γ•E Δ φ' ⊢ θ.
Proof. intro Hp. rewrite E_eq. eapply conjunction_L.
- eapply in_map_in. simpl. intros; apply e_rule_cong; trivial.
- exact Hp.
Qed.

Lemma A_right {Γ} {Δ} {φ φ'} (Hin : φ ∈ Δ) :
Γ ⊢ a_rule (λ (d : env) (_ : d ≺ Δ), E d φ') (λ pe (_ : pe ≺· (Δ, φ')), A pe.1 pe.2) φ Hin ->
Γ ⊢ A Δ φ'.
Proof. intro Hp. rewrite A_eq. eapply make_disj_sound_R, OrR1, disjunction_R.
- eapply in_map_in. simpl. intros; apply a_rule_cong; trivial.
- simpl. exact Hp.
Qed.

Proposition uniform_correct Γ Δ ϕ: (∀ φ0, φ0 ∈ Γ -> ¬ occurs_in p φ0) -> (Γ ⊎ Δ ⊢ ϕ) ->
  (¬ occurs_in p ϕ -> Γ•E Δ ϕ ⊢ ϕ) *
  (Γ•E Δ ϕ ⊢ A Δ ϕ).
Proof.

Section UniformInterpolation.

Uniform interpolation, corollaries

We now show the uniform interpolation as usually stated, which is a corollary of our work above.

(* The uniform interpolants with respect to a list of variables are computed by consecutive application. *)
Definition E_of_list (V : list variable) (φ : formula) : formula :=
fold_right (λ p, λ ψ, E p {[ ψ ]} ⊥) φ V.

Definition A_of_list (V : list variable) (φ : formula) : formula :=
fold_right (λ p, λ ψ, A p ∅ ψ) φ V.

Definition has_no_vars_from θ (V : list variable) : Prop :=
  (∀ p, occurs_in p θ -> ¬ (In p V)).

Local Coercion singleton : formula >-> env.

Theorem E_of_list_correct :
 ∀ V φ, let θ := E_of_list V φ in
 (∀ p, (occurs_in p θ) -> (occurs_in p φ)) *
 (has_no_vars_from θ V) *
 (φ ⊢ θ) *
 (∀ ψ, has_no_vars_from ψ V -> φ ⊢ ψ -> θ ⊢ ψ).
Proof.
  induction V as [|p ps IH].
  - unfold E_of_list; simpl; intro φ; repeat split.
    + tauto.
    + unfold has_no_vars_from. tauto.
    + peapply (generalised_axiom ∅ φ).
    + tauto.
  - intro φ; specialize (IH φ); simpl in IH; repeat split.
    + intros q Hq. unfold E_of_list in Hq.
      simpl in Hq.
      fold (E_of_list ps φ) in Hq.
      apply EA_vars in Hq.
      destruct Hq as [_ [θ [Hin Hθ]]].
      assert (θ = E_of_list ps φ) by ms.
      subst θ.
      apply IH.1.1.1.
      exact Hθ.
    + unfold has_no_vars_from. intros q Hq [Heq|HqIn].
      * subst q. apply EA_vars in Hq. tauto.
      * unfold E_of_list in Hq. simpl in Hq.
        fold (E_of_list ps φ) in Hq.
        apply EA_vars in Hq as [_ [θ [Hin Hθ]]].
        assert (θ = E_of_list ps φ) by ms. subst θ.
        apply IH.1.1.2 in Hθ.
        exact (Hθ HqIn).
    + admit. (* requires cut? *)
    + admit. (* from uniform_correct *)
Admitted.
